    - Избежание избыточных вычислений внутри циклов.

## Разбор кода решения задачи

```c
#include <stdio.h> // Подключение библиотеки для ввода/вывода
#include <windows.h> // Подключение модуля windows.h для установки кодировки вывода

int main() { // Объявляем главную функцию
    SetConsoleOutputCP(CP_UTF8); // Устанавливаем кодировку UTF-8 для корректного отображения русских символов

    double A, result = 1.0; // Объявляем переменные для числа A и результата вычисления степени
    int N, exponent; // Объявляем переменные для показателя степени и счетчика цикла

    printf("Введите число A: "); // Запрашиваем ввод числа A
    scanf("%lf", &A); // Считываем значение A
    printf("Введите натуральный показатель степени N: "); // Запрашиваем ввод показателя степени N
    scanf("%d", &N); // Считываем значение N

    if(N < 1) { // Проверяем, что N является натуральным числом
        printf("Ошибка: показатель степени должен быть натуральным числом (N >= 1).\n"); // Выводим сообщение об ошибке
        return 1; // Завершаем программу с кодом ошибки
    }

    for(exponent = 1; exponent <= N; exponent++) { // Запускаем цикл от 1 до N для вычисления степени
        result *= A; // Умножаем результат на A в каждой итерации
    }

    printf("%.2lf в степени %d равно %.2lf\n", A, N, result); // Выводим результат вычисления степени

    return 0; // Завершаем программу
}
```

> Если Вы используете не Windows, а, например, macOS, то проблем с кодировкой быть не должно — в таком случае игнорируйте следующие строки:
