3. **Системы счисления**
   - Двоичная и восьмеричная системы счисления.
   - Преобразование чисел из одной системы счисления в другую.
4. **Условные операторы**

6. **Строковые функции**
   - Работа с строками при необходимости обработки чисел в системах счисления, отличных от десятичной.


## Разбор кода решения задачи

```c
#include <stdio.h> // Подключение библиотеки для ввода/вывода
#include <windows.h> // Подключение модуля windows.h для установки кодировки вывода
#include <string.h> // Подключение библиотеки для работы со строками

int main() { // Объявляем главную функцию
    SetConsoleOutputCP(CP_UTF8); // Устанавливаем кодировку UTF-8 для корректного отображения русских символов

    char binary[65]; // Объявляем строку для хранения двоичного числа (до 64 бит)
    long long decimal = 0; // Инициализируем переменную для хранения десятичного эквивалента

    printf("Введите натуральное число N в двоичной системе (до 64 бит): "); // Запрашиваем ввод двоичного числа
    scanf("%s", binary); // Считываем двоичное число как строку

    int len = strlen(binary); // Определяем длину введенной строки
    for(int i = 0; i < len; i++) { // Запускаем цикл по каждой цифре двоичного числа
        if(binary[i] != '0' && binary[i] != '1') { // Проверяем, является ли символ цифрой '0' или '1'
            printf("Ошибка: введено не двоичное число.\n"); // Выводим сообщение об ошибке при некорректном вводе
            return 1; // Завершаем программу с кодом ошибки
        }
        decimal = decimal * 2 + (binary[i] - '0'); // Преобразуем двоичную цифру в десятичную и добавляем к общей сумме
    }

    printf("Эквивалент числа %s в десятичной системе: %lld\n", binary, decimal); // Выводим результат преобразования

    return 0; // Завершаем программу
}
```
